EVENEMENT : 
	SELECT * FROM evenement WHERE nom = "Festival Artefact"
	
EVENEMENT par theme : 
	SELECT * FROM evenement WHERE theme = "festival"
				
EVENEMENT PAR VILLE : 
	SELECT * FROM evenement WHERE ville = "strasbourg"
	
TOUT LES EVENEMENTS ordonnée par ville : 
	SELECT * FROM evenement ORDER BY ville asc
	
TOUT LES COIVOTURAGE POUR UN FESTIVAL DONNE
	SELECT covoiturage.date, covoiturage.villedepart, covoiturage.villearrivee FROM evenement,covoiturage WHERE covoiturage.id_event = evenement.id_event AND nom = "Festival Artefact" ORDER BY covoiturage.villedepart asc

TOUT LES COUCHSURFING POUR UN FESTIVAL DONNE
	SELECT * FROM evenement,couchsurfing WHERE couchsurfing.id_event = evenement.id_event AND nom = 'Festival hip-op'
	
PERSONNE PAR INTERET 
	SELECT prenom,nom,ville,interet
	FROM utilisateurservice, utilisateurinfo, utilisateur
	WHERE utilisateur.id_info = utilisateurinfo.id_info AND utilisateur.id_service = utilisateurservice.id_service AND interet LIKE '%musique%'
